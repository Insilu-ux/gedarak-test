function searchFAQ(){

let search =
document.getElementById("helpSearch")
.value
.toLowerCase();

let questions =
document.querySelectorAll(".faq");

let found = false;

questions.forEach(function(item){

let question =
item.querySelector(".faqQuestion");

let answer =
item.querySelector(".faqAnswer");

question.innerHTML =
question.innerHTML.replace(/<mark>|<\/mark>/g,"");

answer.innerHTML =
answer.innerHTML.replace(/<mark>|<\/mark>/g,"");

let qText =
question.innerText
.toLowerCase();

let aText =
answer.innerText
.toLowerCase();

let searchWords =
search
.replace("sharing","share")
.replace("shared","share")
.replace("save","favourite")
.replace("saved","favourite")
.replace("call","contact")
.replace("phone","contact");

let words =
searchWords.split(" ");

let matched = words.every(function(word){

return qText.includes(word) ||
aText.includes(word);

});

if(matched){

item.style.display="block";

found = true;

if(search!=""){

question.innerHTML =
question.innerHTML.replace(
new RegExp(search,"gi"),
match=>"<mark>"+match+"</mark>"
);

answer.innerHTML =
answer.innerHTML.replace(
new RegExp(search,"gi"),
match=>"<mark>"+match+"</mark>"
);

}

}else{

item.style.display="none";

}

});

document.getElementById("noResults").style.display =
found ? "none" : "block";

if(search!=""){

let first =
document.querySelector(
'.faq[style*="block"] .faqQuestion'
);

if(first){

let answer =
first.nextElementSibling;

answer.style.maxHeight =
answer.scrollHeight+"px";

}

}

document.getElementById("suggestions").style.display = "none";
updateSuggestions();

}

box.style.display="none";

}

document.querySelectorAll(".faqQuestion")

let text =
q.innerText.toLowerCase();

if(

text.includes(search)

||

(search=="share" &&
text.includes("sharing"))

||

(search=="contact" &&
text.includes("seller"))

||

(search=="save" &&
text.includes("favourite"))

){

list.push(q.innerText);

}{

list.push(q.innerText);

}

});

if(list.length==0){

return;

}

box.style.display="block";

    function intelligentSearch(){

let text =
document.getElementById("searchInput")
.value
.toLowerCase();
text = text 
.replace(/\bhouses\b/g,"house")
.replace(/\bvillas\b/g,"villa")
.replace(/\bapartments\b/g,"apartment")
.replace(/\boffices\b/g,"office")
.replace(/\bhotels\b/g,"hotel")
.replace(/\bshops\b/g,"shop")
.replace(/\bwarehouses\b/g,"warehouse")
.replace(/\bbuildings\b/g,"building")
.replace(/\bbungalows\b/g,"bungalow")
.replace(/\bannexes\b/g,"annex")
.replace(/\blands\b/g,"land");

// Reset all automatic filters

selectedBedroom = "";

selectedBathroom = "";

document.getElementById("typeFilter").value = "";

document.getElementById("cityFilter").value = "";

document.getElementById("rentOnlyFilter").checked = false;

document.getElementById("maxPrice").value = "";
document.getElementById("amenityFilter").value = "";
document.getElementById("propertyViewFilter").value = "";
document.getElementById("districtFilter").value = "";
document.getElementById("minPrice").value = "";
if(text.trim()==""){

    searchProperty();

    return;

}
let bedroomMatch =
text.match(/([1-9])\s*(bed|bedroom|bedrooms)/);

if(bedroomMatch){

selectedBedroom =
bedroomMatch[1];

}
let bathroomMatch =
text.match(/([1-9])\s*(bath|bathroom|bathrooms)/);

if(bathroomMatch){

selectedBathroom =
bathroomMatch[1];

}

house:"House",

building:"Building",
annex:"Annex",
commercial:"Commercial",
bungalow:"Bungalow"

};

const words = text
.replace(/[^\w\s]/g,"")
.split(/\s+/);

for(let word of words){

    if(word.endsWith("s")){

        word = word.slice(0,-1);

    }

    if(propertyMap[word]){

        document.getElementById("typeFilter").value =
        propertyMap[word];

        break;

    }

}

        function timeAgo(date){

    let now = new Date();

    let posted = new Date(date);

    let diff = now - posted;

    let days = Math.floor(diff/(1000*60*60*24));

    if(days===0){
        return "Today";
    }

    if(days===1){
        return "Yesterday";
    }

    if(days<7){
        return days + " days ago";
    }

    if(days<30){
        return Math.floor(days/7) + " week(s) ago";
    }

    return posted.toLocaleDateString();

}

function score(boost){

if(!boost) return 0;

if(boost.status !== "Active") return 0;

if(boost.endDate < Date.now()) return 0;

if(boost.package === "Platinum") return 3;

if(boost.package === "Premium") return 2;

return 0;

}

function getBoostInfo(property){

if(
!property.boost ||
property.boost.status !== "Active"
){
return null;
}

let remainingMs =
property.boost.endDate - Date.now();

if(remainingMs <= 0){
return null;
}

let daysLeft =
Math.ceil(remainingMs / (1000 * 60 * 60 * 24));

return {
package: property.boost.package,
daysLeft: daysLeft
};
}

function selectBoost(packageName, price, days){

let propertyId =
localStorage.getItem("lastPostedProperty");

if(!propertyId){
    alert("Property not found.");
    return;
}

for(let i=0;i<localStorage.length;i++){

    let key = localStorage.key(i);

    try{

        let property =
        JSON.parse(localStorage.getItem(key));

        if(
            property &&
            String(property.id) === String(propertyId)
        ){

            property.boost = {

                package: packageName,

                price: price,

                days: days,

                status: "Active",

                startDate: Date.now(),

                endDate:
                Date.now() +
                (days*24*60*60*1000)

            };

            localStorage.setItem(
                key,
                JSON.stringify(property)
            );

            break;

        }

    }catch(e){}

}

alert(
"✅ "+packageName+
" Boost Activated!"
);

skipBoost();

}

.map(function(p){

    let score = 0;

    // Same Property Type
    if(p.propertyType === currentProperty.propertyType){
        score += 25;
    }

    // Same District
    if(p.district === currentProperty.district){
        score += 20;
    }

    // Same City
    if(p.city === currentProperty.city){
        score += 30;
    }

    // Same Status
    if(p.status === currentProperty.status){
        score += 15;
    }

    // Bedrooms
    if(Number(p.bedrooms) === Number(currentProperty.bedrooms)){
        score += 12;
    }

    // Bathrooms
    if(Number(p.bathrooms) === Number(currentProperty.bathrooms)){
        score += 10;
    }

    // Price (within 20%)
    if(
        Math.abs(p.price-currentProperty.price)
        <= currentProperty.price*0.20
    ){
        score += 20;
    }

    // Land Size (Perches)
    if(
        Math.abs(p.perches-currentProperty.perches)
        <= 5
    ){
        score += 5;
    }

    // Building Size
    if(
        Math.abs(p.squareFeet-currentProperty.squareFeet)
        <= 500
    ){
        score += 5;
    }

    // Property View
    if(
        (p.propertyView||[]).some(v =>
            (currentProperty.propertyView||[]).includes(v)
        )
    ){
        score += 4;
    }
     // Property Amenities
    if(
        (p.amenities||[]).some(v =>
            (currentProperty.amenities||[]).includes(v)
        )
    ){
        score += 3;
    }

    return {
        property:p,
        score:score
    };

})
.sort(function(a,b){

    return b.score-a.score;

})

    return item.score >= 50;

    function toggleFavourite(id, heart){

                localStorage.getItem(
                    "favourites_" + currentUser
                ) || "[]"
            );

                favourites = 
                favourites.filter(

                heart.style.color = "white";

                heart.style.color = "red";
                showNotification(
                "✔ Saved: " + property.name
               )

      function timeAgo(date){

    let diff = now - posted;

    let days = Math.floor(diff / (1000*60*60*24));

    if(days === 0){
        return "Today";
    }

    if(days === 1){
        return "Yesterday";
    }

    if(days < 7){
        return days + " days ago";
    }

    if(days < 30){
        return Math.floor(days/7) + " week(s) ago";
    }

    return posted.toLocaleDateString();

      }
